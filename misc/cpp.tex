%%% depends on ifempty.tex

\newcommand{\cppmacro}[1]{\LUcode{#1}}
\newcommand{\jppmacro}[1]{\LUcode{#1}}
%%% These macros are supposed to be used as a hook to change styling of keywords via \renewcommand.
\newcommand{\cppmacrokeyword}[1]{\##1}
\newcommand{\jppmacrokeyword}[1]{//\##1}

\newcommand{\cppdirective}[2]{\cppmacro{\cppmacrokeyword{#1}\ifnotempty{#2}{ #2}}}
\newcommand{\jppdirective}[2]{\jppmacro{\jppmacrokeyword{#1}\ifnotempty{#2}{ #2}}}

%%% Alternatives for \cppdirective and \jppdirective that can be used when the second argument is not empty.
%%% Use these macros in case something breaks with the inner \ifnotempty within \cppdirective/\jppdirective.
\newcommand{\cppdirectivei}[2]{\cppmacro{\cppmacrokeyword{#1} #2}}
\newcommand{\jppdirectivei}[2]{\jppmacro{\jppmacrokeyword{#1} #2}}

\newcommand{\cppif}[1]{\cppdirective{if}{#1}}
\newcommand{\cppifdef}[1]{\cppdirective{ifdef}{#1}}
\newcommand{\cppelse}[0]{\cppdirective{else}{}}
\newcommand{\cppelif}[0]{\cppdirective{elif}{}}
\newcommand{\cppifndef}[1]{\cppdirective{ifndef}{#1}}
\newcommand{\cppendif}[0]{\cppdirective{endif}{}}
\newcommand{\cppinclude}[1]{\cppdirective{include}{#1}}
\newcommand{\cppifndefbold}[1]{\cppdirective{if\textbf{n}def}{#1}}

\newcommand{\jppif}[1]{\jppdirective{if}{#1}}

\newcommand{\PP}[0]{preprocessor}
\newcommand{\PPCaps}[0]{Preprocessor}
\newcommand{\PPTitleCase}[0]{Preprocessor}
\makePluralMacro{PP}
\makePluralMacro{PPCaps}
\makePluralMacro{PPTitleCase}

\newcommand{\CPP}[0]{C \PP}
\newcommand{\CPPCaps}[0]{C \PP}
\newcommand{\CPPTitleCase}[0]{C \PPTitleCase}

\newcommand{\JPP}[0]{Java \PP}
\newcommand{\JPPCaps}[0]{Java \PP}
\newcommand{\JPPTitleCase}[0]{Java \PPTitleCase}
