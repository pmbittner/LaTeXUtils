\newcommand{\cppendif}[0]{\cppmacro{\cppmacrokeyword{endif}}}
\newcommand{\cppinclude}[1]{\cppmacro{\cppmacrokeyword{include}\ifnotempty{#1}{ #1}}}
\newcommand{\cppifndefbold}[1]{\cppmacro{\cppmacrokeyword{if\textbf{n}def}\ifnotempty{#1}{ #1}}}

\newcommand{\jppif}[1]{\jppmacro{\jppmacrokeyword{if}\ifnotempty{#1}{ #1}}}
