%--------------------------

\newlength{\LUCBWidth}%

\newcommand{\LUColorBox}[2][red]%
{%
	\settowidth{\LUCBWidth}{#2}%
	%\setlength{\fboxsep}{0pt}%
	\colorbox{#1}%
	{%      
		\raisebox{-\LUCBDepthReference}%
		{%
			\parbox[b][\LUCBHeightReference+\LUCBDepthReference][c]{\LUCBWidth}{\centering#2}%
		}%
	}%
}
