\newcommand{\latexutilautorefoverride}[2]{
	\ifcsname #1autorefname\endcsname
		\expandafter\renewcommand\csname #1autorefname\endcsname{#2}
	\else
		\expandafter\newcommand\csname #1autorefname\endcsname{#2}
	\fi	
}
