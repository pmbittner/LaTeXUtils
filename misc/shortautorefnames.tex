\newcommand{\latexutilautorefoverride}[2]{
	\ifcsname #1\endcsname
		\expandafter\renewcommand\csname #1\endcsname{#2}
	\else
		\expandafter\newcommand\csname #1\endcsname{#2}
	\fi	
}
